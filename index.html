document.getElementById('btnCreate').onclick = async () => { // 1. Deteksi Library dengan Fallback (Sesuai Foto Error) const web3 = window.solanaWeb3; const spl = window.splToken || window.solanaSplToken || (window.solanaSplToken ? window.solanaSplToken : null); const mpl = window.mplTokenMetadata; if (!web3 || !spl) { alert("Sabar Bro, library Solana lagi loading... Coba klik lagi dalam 3 detik."); return; } if (isLaunching) return; // 2. Ambil semua input (Pastikan ID di HTML lo sesuai) const name = document.getElementById('name').value.trim(); const symbol = document.getElementById('symbol').value.trim(); const desc = document.getElementById('desc').value.trim(); const supplyRaw = document.getElementById('supply').value; const logoFile = document.getElementById('logoFile').files[0]; // Ambil input Social Media (Jangan sampai hilang) const web = document.getElementById('website').value.trim(); const tw = document.getElementById('twitter').value.trim(); const tg = document.getElementById('telegram').value.trim(); // 3. Validasi Input if (!userPubKey) return alert("Hubungkan Wallet Phantom dulu!"); if (!name || !symbol || !supplyRaw || !logoFile) return alert("Wajib isi: Nama, Simbol, Supply, dan Logo!"); isLaunching = true; const btn = document.getElementById('btnCreate'); btn.disabled = true; btn.innerText = "SEDANG MEMPROSES..."; try { const supplyVal = BigInt(supplyRaw); const decimals = parseInt(document.getElementById('decimals').value) || 9; const connection = new web3.Connection(HELIUS_RPC, "confirmed"); const METADATA_PROGRAM_ID = new web3.PublicKey("metaqbxxUfaypS9e3uJ8teU67cKxSBy79z3G2v4f99T"); updateLog("Step 1: Uploading Metadata ke IPFS..."); const metadataUri = await uploadIPFS(logoFile, name, symbol, desc, web, tw, tg); updateLog("Step 2: Menyiapkan Transaksi Blockchain..."); const mintKey = web3.Keypair.generate(); const transaction = new web3.Transaction(); const { blockhash } = await connection.getLatestBlockhash(); transaction.recentBlockhash = blockhash; transaction.feePayer = userPubKey; // A. Transfer Fee & Buat Account Mint transaction.add( web3.SystemProgram.transfer({ fromPubkey: userPubKey, toPubkey: new web3.PublicKey(OWNER_FEE_DEST), lamports: 0.3 * web3.LAMPORTS_PER_SOL }), web3.SystemProgram.createAccount({ fromPubkey: userPubKey, newAccountPubkey: mintKey.publicKey, space: 82, lamports: await connection.getMinimumBalanceForRentExemption(82), programId: spl.TOKEN_PROGRAM_ID // FIX: Pakai spl yang terdeteksi }), spl.createInitializeMintInstruction(mintKey.publicKey, decimals, userPubKey, userPubKey) ); // B. Tambahkan Metadata Metaplex (Agar Logo & Nama Muncul) const [metadataPDA] = web3.PublicKey.findProgramAddressSync( [Buffer.from("metadata"), METADATA_PROGRAM_ID.toBuffer(), mintKey.publicKey.toBuffer()], METADATA_PROGRAM_ID ); if (mpl) { transaction.add( mpl.createCreateMetadataAccountV3Instruction( { metadata: metadataPDA, mint: mintKey.publicKey, mintAuthority: userPubKey, payer: userPubKey, updateAuthority: userPubKey }, { createMetadataAccountV3Args: { data: { name, symbol, uri: metadataUri, sellerFeeBasisPoints: 0, creators: null, collection: null, uses: null }, isMutable: false, collectionDetails: null } } ) ); } // C. Buat ATA & Mint Supply ke Wallet User const ata = await spl.getAssociatedTokenAddress(mintKey.publicKey, userPubKey); transaction.add( spl.createAssociatedTokenAccountInstruction(userPubKey, ata, userPubKey, mintKey.publicKey), spl.createMintToInstruction(mintKey.publicKey, ata, userPubKey, supplyVal * (BigInt(10) ** BigInt(decimals))), // D. Revoke Authority (Rug-Proof: Mint & Freeze dimatikan) spl.createSetAuthorityInstruction(mintKey.publicKey, userPubKey, spl.AuthorityType.MintTokens, null), spl.createSetAuthorityInstruction(mintKey.publicKey, userPubKey, spl.AuthorityType.FreezeAccount, null) ); updateLog("Step 3: Silakan ACC di Wallet Phantom lo..."); transaction.partialSign(mintKey); const provider = window?.phantom?.solana || window?.solana; const signed = await provider.signTransaction(transaction); const signature = await connection.sendRawTransaction(signed.serialize()); updateLog("Step 4: Menunggu Konfirmasi Jaringan..."); await connection.confirmTransaction(signature); const ca = mintKey.publicKey.toBase58(); updateLog(`BERHASIL!`); updateLog(`CA: ${ca}`); alert(`GOKIL! Token Berhasil Dibuat.\nCA: ${ca}`); window.open(`https://solscan.io/token/${ca}`, '_blank'); } catch (err) { updateLog(`Gagal: ${err.message}`); alert("Error: " + err.message); } finally { isLaunching = false; btn.disabled = false; btn.innerText = "LAUNCH TOKEN (0.3 SOL)"; } };
